$$
\begin{gather}
\frac{1}{x-2}+\frac{x-1}{x+2}=1\\
\textcolor{RoyalBlue}{\frac{x+2}{x+2}} \frac{1}{\textcolor{PineGreen}{x-2}}+\textcolor{PineGreen}{\frac{x-2}{x-2}} \frac{x-1}{\textcolor{RoyalBlue}{x+2}}=1
\end{gather}
$$
